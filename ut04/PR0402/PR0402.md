```python
def contar_vocales_consonantes(cadena):
    vocales = "aeiou"
    consonantes = "bcdfghjklmnñpqrstvwxyz"
    
    contador_vocales = 0
    contador_consonantes = 0
    
    cadena_minusculas = cadena.lower()
    
    for caracter in cadena_minusculas:
        if caracter in vocales:
            contador_vocales += 1
        elif caracter in consonantes:
            contador_consonantes += 1
            
    return contador_vocales, contador_consonantes

cadena_del_usuario = input("Introduce la cadena de texto: ")

vocales, consonantes = contar_vocales_consonantes(cadena_del_usuario)

print(f"Cadena introducida: '{cadena_del_usuario}'")
print(f"Número de vocales: {vocales}")
print(f"Número de consonantes: {consonantes}")
```

```python
def eliminar_duplicados_interactivo():

  cadena = input("Introduce la cadena de texto para eliminar sus caracteres duplicados: ")
  
  caracteres_vistos = set()
  resultado = []
  
  for caracter in cadena:
    
    if caracter not in caracteres_vistos:
     
      resultado.append(caracter)
    
      caracteres_vistos.add(caracter)
      
  cadena_sin_duplicados = "".join(resultado)
  
  print(f"Cadena original: '{cadena}'")
  print(f"Cadena sin duplicados: '{cadena_sin_duplicados}'")
 


eliminar_duplicados_interactivo()

```

```python
def rotar_izquierda_interactivo():

  cadena = input("Introduce la cadena de texto a rotar: ")
  
  while True:
    try:
      
      n_str = input("Introduce el número de posiciones para rotar a la izquierda: ")
      n = int(n_str)
      break  
    except ValueError:
      print("Error, introduce un número entero válido.")

  longitud = len(cadena)
  
  if longitud == 0:
    print("La cadena está vacía.")
    return

  n_efectivo = n % longitud
  

  parte_a_mover = cadena[:n_efectivo]
  

  parte_inicial = cadena[n_efectivo:]
  

  cadena_rotada = parte_inicial + parte_a_mover
  
 
  print(f"Cadena rotada:      '{cadena_rotada}'")
  

rotar_izquierda_interactivo()
```

```python
def codificar_rle(cadena):
    if not cadena:  
        return ""
    
    resultado = ""    
    contador = 1     
    
    for i in range(1, len(cadena)):
        if cadena[i] == cadena[i - 1]:
            contador += 1  
        else:
            
            resultado += cadena[i - 1] + str(contador)
            contador = 1  
    
    
    resultado += cadena[-1] + str(contador)
    
    return resultado


texto = "luuiiissss"
print(codificar_rle(texto)) 
```
