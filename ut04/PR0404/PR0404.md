### Ejercicio 4
```python
asignaturas = {
    "Matemáticas": ["Ana", "Carlos", "Luis", "María", "Jorge"],
    "Física": ["Elena", "Luis", "Juan", "Sofía"],
    "Programación": ["Ana", "Carlos", "Sofía", "Jorge", "Pedro"],
    "Historia": ["María", "Juan", "Elena", "Ana"],
    "Inglés": ["Carlos", "Sofía", "Jorge", "María"],
}

def listar(asignatura):
    if asignatura in asignaturas:
        print(f"\nEstudiantes en {asignatura}:")
        for estudiante in asignaturas[asignatura]:
            print(f"- {estudiante}")
    else:
        print("La asignatura no existe.")

def matricular(asignatura, estudiante):
    if asignatura in asignaturas:
        if estudiante not in asignaturas[asignatura]:
            asignaturas[asignatura].append(estudiante)
            print(f"{estudiante} matriculado en {asignatura}.")
        else:
            print(f"{estudiante} ya está matriculado en {asignatura}.")
    else:
        print("La asignatura no existe.")

def baja(asignatura, estudiante):
    if asignatura in asignaturas:
        if estudiante in asignaturas[asignatura]:
            asignaturas[asignatura].remove(estudiante)
            print(f"{estudiante} dado de baja en {asignatura}.")
        else:
            print(f"{estudiante} no está matriculado en {asignatura}.")
    else:
        print("La asignatura no existe.")


while True:
    print("1. Listar estudiantes de una asignatura")
    print("2. Matricular estudiante")
    print("3. Dar de baja estudiante")
    print("4. Salir")

    opcion = input("Elige opción: ")

    if opcion == "1":
        asignatura = input("Asignatura: ")
        listar(asignatura)

    elif opcion == "2":
        asignatura = input("Asignatura: ")
        estudiante = input("Nombre del estudiante: ")
        matricular(asignatura, estudiante)

    elif opcion == "3":
        asignatura = input("Asignatura: ")
        estudiante = input("Nombre del estudiante: ")
        baja(asignatura, estudiante)

    elif opcion == "4":
        print("Saliendo...")
        break

    else:
        print("Opción no válida.")
```

def diccionario_invertido(dic):
    invertido = {}

        invertido[valor] = clave
    return invertido

def combinar_diccionarios(dic1, dic2):
    resultado = {}

    
    for producto, precio in dic1.items():
        resultado[producto] = precio

   
    for producto, precio in dic2.items():
        if producto in resultado:
            resultado[producto] += precio 
        else:
            resultado[producto] = precio   

    return resultado


productos1 = {
    "Manzana": 1.2,
    "Banana": 0.8,
    "Naranja": 1.0
}

productos2 = {
    "Banana": 1.0,
    "Pera": 1.5,
    "Naranja": 0.9
}

resultado = combinar_diccionarios(productos1, productos2)
print(resultado)
