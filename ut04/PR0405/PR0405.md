### Ejercicio 2

### Ejercicio 5
```python
from functools import reduce


numeros = [1, 2, 3, 4, 5, 6]


pares = filter(lambda x: x % 2 == 0, numeros)


pares_mapeados = map(lambda x: x, pares)


producto = reduce(lambda x, y: x * y, pares_mapeados, 1)

print("Producto de números pares:", producto)
```
### Ejercicio 6
```python
from functools import reduce

numeros = [[-3, 2, 7], [10, -5, 3], [0, 8, -2]]


aplanado = reduce(lambda x, y: x + y, numeros)


positivos = filter(lambda n: n > 0, aplanado)


resultado = reduce(lambda x, y: x + y, positivos, 0)

print("Suma de positivos:", resultado)
```
