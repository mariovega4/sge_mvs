### Ejercicio 6
```python
UNIDADES = {
    'mm': 0.001,
    'cm': 0.01,
    'm': 1.0,
    'km': 1000.0
}

UNIDADES_VALIDAS = UNIDADES.keys()
print("--- Conversor de Longitud (mm, cm, m, km) ---")


while True:
    try:
        cantidad_input = input("Introduce la cantidad a convertir: ")
        cantidad = float(cantidad_input)
        if cantidad >= 0:
            break
        print("No se pueden introducir numeros negativos.")
    except ValueError:
        print("Entrada no válida. Introduce un número.")


print(f"\nUnidades disponibles: {', '.join(UNIDADES_VALIDAS)}")
while True:
    unidad_origen = input("Unidad de medida de origen (ej. m): ").lower().strip()
    if unidad_origen in UNIDADES_VALIDAS:
        break
    print(f"Unidad no válida. Por favor, usa una de estas: {', '.join(UNIDADES_VALIDAS)}")

while True:
    unidad_destino = input("Unidad de medida a convertir (ej. cm): ").lower().strip()
    if unidad_destino in UNIDADES_VALIDAS:
        break
    print(f"Unidad no válida. Por favor, usa una de estas: {', '.join(UNIDADES_VALIDAS)}")

factor_origen = UNIDADES[unidad_origen]
cantidad_en_m = cantidad * factor_origen

factor_destino_a_m = UNIDADES[unidad_destino]

factor_de_m_a_destino = 1 / factor_destino_a_m

resultado = cantidad_en_m * factor_de_m_a_destino

print("\n" + "="*50)
print(f" Resultado: {cantidad:,.4f} {unidad_origen} son {resultado:,.4f} {unidad_destino}")
print("="*50)
```

```python
print("--- Generador de Secuencia de Fibonacci ---")

while True:
    try:
        num_n_input = input("¿Cuántos números de la secuencia de Fibonacci quieres generar? (n > 0): ")
        num_n = int(num_n_input)
        if num_n > 0:
            break
        else:
            print("Por favor, introduce un entero positivo (n > 0).")
    except ValueError:
        print("Entrada no válida. Por favor, introduce un número entero.")

if num_n == 1:
    secuencia = [1]
elif num_n == 2:
    secuencia = [1, 1]
else:   
    
    a = 1  
    b = 1  
    
    secuencia = [a, b]

    for _ in range(2, num_n):
        
        siguiente = a + b
        
        secuencia.append(siguiente)
        
        a = b
        b = siguiente

print(f"\nLos primeros {num_n} números de Fibonacci son:")
print(", ".join(map(str, secuencia)))
print("---------------------------------------------")
```

print("--- Verificador de Número Primo ---")

while True:
    try:
        num_input = input("Introduce un número entero positivo para verificar si es primo: ")
        numero = int(num_input)
        if numero > 0:
            break
        else:
            print("Por favor, introduce un número entero positivo.")
    except ValueError:
        print("Entrada no válida. Debes introducir un número entero.")

es_primo = True

if numero == 1:
    
    es_primo = False
elif numero == 2:
    
    es_primo = True
else:

    for i in range(2, numero):
        if numero % i == 0:
            
            es_primo = False
            break 

print(f"Número introducido: {numero}")

if es_primo:
    print(f"{numero} es un número primo")
else:
    print(f"{numero} no es un número primo")
    if numero > 1:
        print(f"Es divisible por otros números además de 1 y {numero}.")
